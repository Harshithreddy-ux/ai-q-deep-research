wcss = []
for i in range(1, 11):
    kmeans = KMeans(n_clusters=i, init='k-means++', random_state=42, n_init=10)
    kmeans.fit(X_scaled)

X_train_A, X_test_A, y_train_A, y_test_A = train_test_split(X_scaled, y_data, test_size=0.2, random_state=42)

# Define Models
rf_A = RandomForestClassifier(n_estimators=200, max_depth=10, random_state=42)
xgb_A = xgb.XGBClassifier(n_estimators=200, learning_rate=0.05, max_depth=5, random_state=42)

# FIX: Fit them individually first so they are "ready" for the results table
rf_A.fit(X_train_A, y_train_A)
xgb_A.fit(X_train_A, y_train_A)

# Define and fit the Ensemble
ensemble_A = VotingClassifier(estimators=[('rf', rf_A), ('xgb', xgb_A)], voting='soft')
ensemble_A.fit(X_train_A, y_train_A)

results_A = [
    {"Model": "RF (No Cluster)", "Train_Acc": rf_A.score(X_train_A, y_train_A), "Test_Acc": rf_A.score(X_test_A, y_test_A)},
    {"Model": "XGB (No Cluster)", "Train_Acc": xgb_A.score(X_train_A, y_train_A), "Test_Acc": xgb_A.score(X_test_A, y_test_A)},
    {"Model": "Ensemble (No Cluster)", "Train_Acc": ensemble_A.score(X_train_A, y_train_A), "Test_Acc": ensemble_A.score(X_test_A, y_test_A)}
